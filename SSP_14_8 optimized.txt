f_hma(src, n) =>
    int _sqn = math.round(math.sqrt(n))
    ta.wma(2.0 * ta.wma(src, math.round(n / 2)) - ta.wma(src, n), math.max(_sqn, 2))

// Normalises any series to approx [-1, +1] using rolling 50-bar mean/stdev.
// Values beyond ±3σ are clamped so a single outlier cannot dominate distance.
f_zscore(_val, _len) =>
    float _m = ta.sma(_val, _len)
    float _s = ta.stdev(_val, _len)
    _s > 0.001 ? math.max(math.min((_val - _m) / _s, 3.0), -3.0) / 3.0 : 0.0

    array.remove(p2_fC,  _ei), array.remove(p2_fD,  _ei)
    array.remove(p2_fE,  _ei), array.remove(p2_fV,  _ei)
    array.remove(p2_rgm, _ei), array.remove(p2_lbl, _ei)
    array.remove(p2_sig, _ei), array.remove(p2_dir, _ei)
    // Adjust OFRL queue pointers so stored indices remain valid
    if array.size(_p2OfrlIdxQ) > 0
        for _qi = 0 to array.size(_p2OfrlIdxQ) - 1
            int _oi = array.get(_p2OfrlIdxQ, _qi)
            if _oi > _ei

            else if _oi == _ei

// SECTION 5 — f_p2_push(_isLong)
// Appends current bar's 18-feature vector, then applies:
//   (a) per-regime FIFO eviction  — keep ≤ i_p2BktMax per regime
//   (b) global hard-ceiling once  — keep ≤ 36 × i_p2BktMax = 1800
// The lowest-quality non-protected sample is evicted first.
// ════════════════════════════════════════════════════════════════════
f_p2_push(_isLong) =>
    // ── Append all 18 features in index order ───────────────────────
    array.push(p2_f0,  _qf0 )
    array.push(p2_f2,  _qf2 )
    array.push(p2_f5,  _qf5 )
    array.push(p2_f7,  _qf7 )
    array.push(p2_f8,  _qf8 )
    array.push(p2_f9,  _qf9 )
    array.push(p2_f13, _qf13)
    array.push(p2_f15, _qf15)
    array.push(p2_f24, _qfHz)   // Hunt zone stored in p2_f24 slot
    array.push(p2_f25, _qf25)
    array.push(p2_f27, _qf27)
    array.push(p2_f28, _qf28)
    array.push(p2_fA,  _qfA )
    array.push(p2_fB,  _qfB )
    array.push(p2_fC,  _qfC )
    array.push(p2_fD,  _qfD )
    array.push(p2_fE,  _qfE )
    array.push(p2_fV,  _qfV )
    // ── Metadata ────────────────────────────────────────────────────
    array.push(p2_rgm, _p2Regime)
    array.push(p2_lbl, 0.5)        // pending label — OFRL overwrites
    array.push(p2_sig, 0.0)
    array.push(p2_dir, _isLong)    // direction placeholder — OFRL overwrites

    // ── (a) Per-regime eviction ──────────────────────────────────────
    int _bktCount = 0
    int _p2Sz     = array.size(p2_lbl)
    for _bi = 0 to _p2Sz - 1
        if array.get(p2_rgm, _bi) == _p2Regime
            _bktCount += 1
    if _bktCount > i_p2BktMax
        int   _evictIdx   = -1
        float _evictScore = 2.0
        for _bi2 = 0 to array.size(p2_lbl) - 1
            if array.get(p2_rgm, _bi2) == _p2Regime
                float _bl   = array.get(p2_lbl,  _bi2)
                float _evSc = _bl == 0.5 ? 0.51 : _bl
                bool  _prot = _bl >= 1.0 and array.get(p2_sig, _bi2) >= 80.0
                if not _prot and (_evictIdx == -1 or _evSc < _evictScore)
                    _evictIdx   := _bi2
                    _evictScore := _evSc
        if _evictIdx >= 0
            f_p2_remove(_evictIdx)

    // ── (b) Global hard-ceiling (36 × i_p2BktMax = max 1800) ────────
    // [FIX 2]: Only ONE while-loop — the duplicate from v10.0 is removed.
    while array.size(p2_lbl) > 162 * i_p2BktMax
        int   _gIdx   = -1
        float _gScore = 2.0
        int _dynKGD = hurstVal > 0.60 ? 10 : hurstVal < 0.45 ? 3 : i_knnK
        int   _kGD  = _dynKGD
        int[]   _gdIdx  = array.new_int(_kGD,  -1)
        float _gdCachedMaxD = 1e9   // cached heap maximum distance
        int   _gdCachedMaxI = 0     // cached heap maximum index
        // ── Cached heap-max for O(1) insertion check ─────────
        //float _gdCachedMaxD = 1e9
        //int   _gdCachedMaxI = 0
        for _gi = 0 to array.size(p2_lbl) - 1
            float _gl    = array.get(p2_lbl,  _gi)
            float _gEvSc = _gl == 0.5 ? 0.51 : _gl
            bool  _gProt = _gl >= 1.0 and array.get(p2_sig, _gi) >= 80.0
            if not _gProt and (_gIdx == -1 or _gEvSc < _gScore)
                _gIdx   := _gi
                _gScore := _gEvSc
        if _gIdx < 0
            _gIdx := 0
        f_p2_remove(_gIdx)

// ════════════════════════════════════════════════════════════════════
// SECTION 6 — KNN PUSH GATE

// ════════════════════════════════════════════════════════════════════
// SECTION 7 — OFRL: OUTCOME FORWARD RESOLUTION (30 bars lookahead)
// Registers each pushed bar for bidirectional forward scanning.
// After i_ofrlLookFwd bars (default 30), the dominant price movement
// determines the label and direction.  Labels feed the KNN buffer.
// After labelling each resolved bar, online gradient descent updates
// the 18 feature weights via cross-entropy loss.

int _adaptLookFwd = atrRatio > 1.5 ? math.max(8, int(i_ofrlLookFwd * 0.5)) :
                         atrRatio < 0.7 ? int(i_ofrlLookFwd * 1.3) :
                         i_ofrlLookFwd
// ════════════════════════════════════════════════════════════════════
if _strideAllow and array.size(_p2OfrlIdxQ) < 100
    float _p2R = math.max(atr14 * i_slAtr, atr14 * 0.3)
    array.push(_p2OfrlIdxQ,  array.size(p2_lbl) - 1)
    array.push(_p2OfrlEntQ,  close)
    array.push(_p2OfrlRQ,    _p2R)

    array.push(_p2OfrlMFEQ,  0.0)
    array.push(_p2OfrlMAEQ,  0.0)

if array.size(_p2OfrlIdxQ) > 0
    int _p2oi = 0
    while _p2oi < array.size(_p2OfrlIdxQ)
        int   _p2Idx  = array.get(_p2OfrlIdxQ,  _p2oi)
        float _p2Ent  = array.get(_p2OfrlEntQ,  _p2oi)
        float _p2R2   = array.get(_p2OfrlRQ,    _p2oi)
        int   _p2Bars = array.get(_p2OfrlBarsQ, _p2oi)

        // Track bidirectional excursion symmetrically
        float _p2MFE = math.max(array.get(_p2OfrlMFEQ, _p2oi), high - _p2Ent)
        float _p2MAE = math.max(array.get(_p2OfrlMAEQ, _p2oi), _p2Ent - low)
        array.set(_p2OfrlMFEQ, _p2oi, _p2MFE)
        array.set(_p2OfrlMAEQ, _p2oi, _p2MAE)

        bool _p2Expd  = _p2Bars <= 0
        bool _p2SlHit = _p2MAE >= _p2R2   // price fell 1R from entry

        if _p2Idx < 0
            // Invalidated pointer — clean up
            array.remove(_p2OfrlIdxQ, _p2oi), array.remove(_p2OfrlEntQ, _p2oi)
            array.remove(_p2OfrlRQ,   _p2oi), array.remove(_p2OfrlLongQ,_p2oi)
            array.remove(_p2OfrlBarsQ,_p2oi), array.remove(_p2OfrlMFEQ, _p2oi)
            array.remove(_p2OfrlMAEQ, _p2oi)

        else if (_p2SlHit or _p2Expd) and _p2Idx < array.size(p2_lbl)
            // ── Determine actual outcome ────────────────────────────
            float _upR = _p2R2 > 0 ? _p2MFE / _p2R2 : 0.0
            float _dnR = _p2R2 > 0 ? _p2MAE / _p2R2 : 0.0
            bool  _actualBull = _p2MFE > _p2MAE * 1.5 and _p2MAE < _p2R2
            bool  _actualBear = _p2MAE > _p2MFE * 1.5 and _p2MFE < _p2R2
            float _strongerR  = math.max(_upR, _dnR)

            // Quality label — reflects magnitude of directional move
            float _p2Lbl = 0.5   // default: mixed/unclear, skip in KNN
            if _actualBull or _actualBear
                _p2Lbl := _strongerR >= 2.0 ? 1.00 :   // 3R+ clean move
                          _strongerR >= 1.5 ? 0.85 :   // 2R+ good move
                          _strongerR >= 1.0 ? 0.70 :   // 1R+ decent move
                          _strongerR >= 0.5 ? 0.40 :   // sub-1R weak
                                              0.30      // micro-directional

            // [FIX — direction-aware]: store actual direction from OFRL
            if _p2Idx >= 0 and _p2Idx < array.size(p2_dir)
                array.set(p2_dir, _p2Idx, _actualBull)

            // [FIX 3 from v7.7]: unresolved bars (0.5) after 30-bar
            // expiry are set to 0.5 and excluded from KNN voting.
            if _p2Lbl != 0.5
                array.set(p2_lbl, _p2Idx, _p2Lbl)
            else
                array.set(p2_lbl, _p2Idx, 0.5)


            // ═══════════════════════════════════════════════════════
            // ONLINE GRADIENT DESCENT — Gaussian kernel with adaptive σ
            if _p2Lbl != 0.5 and _p2Idx >= 0
                // Resolved bar feature vector
                float rf0  = array.get(p2_f0,  _p2Idx)
                float rf1  = array.get(p2_f2,  _p2Idx)
                float rf2  = array.get(p2_f5,  _p2Idx)
                float rf3  = array.get(p2_f7,  _p2Idx)
                float rf4  = array.get(p2_f8,  _p2Idx)
                float rf5  = array.get(p2_f9,  _p2Idx)
                float rf6  = array.get(p2_f13, _p2Idx)
                float rf7  = array.get(p2_f15, _p2Idx)
                float rf8  = array.get(p2_f24, _p2Idx)
                float rf9  = array.get(p2_f25, _p2Idx)
                float rf10 = array.get(p2_f27, _p2Idx)
                float rf11 = array.get(p2_f28, _p2Idx)
                float rf12 = array.get(p2_fA,  _p2Idx)
                float rf13 = array.get(p2_fB,  _p2Idx)
                float rf14 = array.get(p2_fC,  _p2Idx)
                float rf15 = array.get(p2_fD,  _p2Idx)
                float rf16 = array.get(p2_fE,  _p2Idx)
                float rf17 = array.get(p2_fV,  _p2Idx)

                int _rfRgm = array.get(p2_rgm, _p2Idx)   // regime of this resolved bar
                int _rfTdir  = (_rfRgm / 9)  % 3
                int _rfVol   = (_rfRgm / 3)  % 3
                int _rfMacro = _rfVol == 2 ? 2 : _rfTdir == 2 ? 0 : _rfTdir == 0 ? 1 : 0
                float[] _rfActiveW = _rfMacro == 2 ? featWeights_hvol :
                     _rfMacro == 1 ? featWeights_range : featWeights_trend
              //  // Adaptive K for gradient descent
                int _dynKGD = hurstVal > 0.60 ? 10 : hurstVal < 0.45 ? 3 : i_knnK
                int _kGD  = _dynKGD
                float[] _gdDist = array.new_float(_kGD, 1e9)
                float[] _gdLbl  = array.new_float(_kGD, 0.5)
                bool[]  _gdDir  = array.new_bool (_kGD, false)
                int[]   _gdIdx  = array.new_int  (_kGD, -1)
                float _gdCachedMaxD = 1e9
                int   _gdCachedMaxI = 0

                // 1. Find K nearest labelled neighbours in same regime
                for _gi = 0 to array.size(p2_lbl) - 1
                    if _gi == _p2Idx
                        continue
                    if array.get(p2_lbl, _gi) == 0.5
                        continue
                    if array.get(p2_rgm, _gi) != _rfRgm
                        continue

                    // Compute per‑feature contributions (fdiff) for this neighbour
                    float _fdiff0  = math.log(1.0+math.abs(rf0  - array.get(p2_f0,  _gi))) * _kS0
                    float _fdiff1  = math.log(1.0+math.abs(rf1  - array.get(p2_f2,  _gi))) * _kS1
                    float _fdiff2  = math.log(1.0+math.abs(rf2  - array.get(p2_f5,  _gi))) * _kS2
                    float _fdiff3  = math.log(1.0+math.abs(rf3  - array.get(p2_f7,  _gi))) * _kS3
                    float _fdiff4  = math.log(1.0+math.abs(rf4  - array.get(p2_f8,  _gi))) * _kS4
                    float _fdiff5  = math.log(1.0+math.abs(rf5  - array.get(p2_f9,  _gi))) * _kS5
                    float _fdiff6  = math.log(1.0+math.abs(rf6  - array.get(p2_f13, _gi))) * _kS6
                    float _fdiff7  = math.log(1.0+math.abs(rf7  - array.get(p2_f15, _gi))) * _kS7
                    float _fdiff8  = math.log(1.0+math.abs(rf8  - array.get(p2_f24, _gi))) * _kS8
                    float _fdiff9  = math.log(1.0+math.abs(rf9  - array.get(p2_f25, _gi))) * _kS9
                    float _fdiff10 = math.log(1.0+math.abs(rf10 - array.get(p2_f27, _gi))) * _kS10
                    float _fdiff11 = math.log(1.0+math.abs(rf11 - array.get(p2_f28, _gi))) * _kS11
                    float _fdiff12 = math.log(1.0+math.abs(rf12 - array.get(p2_fA,  _gi))) * _kS12
                    float _fdiff13 = math.log(1.0+math.abs(rf13 - array.get(p2_fB,  _gi))) * _kS13
                    float _fdiff14 = math.log(1.0+math.abs(rf14 - array.get(p2_fC,  _gi))) * _kS14
                    float _fdiff15 = math.log(1.0+math.abs(rf15 - array.get(p2_fD,  _gi))) * _kS15
                    float _fdiff16 = math.log(1.0+math.abs(rf16 - array.get(p2_fE,  _gi))) * _kS16
                    float _fdiff17 = math.log(1.0+math.abs(rf17 - array.get(p2_fV,  _gi))) * _kS17

                    //// Weighted distance
                    float w0=array.get(_rfActiveW,0),w1=array.get(_rfActiveW,1),w2=array.get(_rfActiveW,2),w3=array.get(_rfActiveW,3),w4=array.get(_rfActiveW,4),w5=array.get(_rfActiveW,5)
                    float w6=array.get(_rfActiveW,6),w7=array.get(_rfActiveW,7),w8=array.get(_rfActiveW,8),w9=array.get(_rfActiveW,9),w10=array.get(_rfActiveW,10),w11=array.get(_rfActiveW,11)
                    float w12=array.get(_rfActiveW,12),w13=array.get(_rfActiveW,13),w14=array.get(_rfActiveW,14),w15=array.get(_rfActiveW,15),w16=array.get(_rfActiveW,16),w17=array.get(_rfActiveW,17)
                    float _dGD =
                             w0  * _fdiff0  +
                             w1  * _fdiff1  +
                             w2  * _fdiff2  +
                             w3  * _fdiff3  +
                             w4  * _fdiff4  +
                             w5  * _fdiff5  +
                             w6  * _fdiff6  +
                             w7  * _fdiff7  +
                             w8  * _fdiff8  +
                             w9  * _fdiff9  +
                             w10 * _fdiff10 +
                             w11 * _fdiff11 +
                             w12 * _fdiff12 +
                             w13 * _fdiff13 +
                             w14 * _fdiff14 +
                             w15 * _fdiff15 +
                             w16 * _fdiff16 +
                             w17 * _fdiff17

                    // Insert into top‑K heap (keep the smallest distances)
                    if _dGD < _gdCachedMaxD
                        array.set(_gdDist, _gdCachedMaxI, _dGD)
                        array.set(_gdLbl,  _gdCachedMaxI, array.get(p2_lbl, _gi))
                        array.set(_gdDir,  _gdCachedMaxI, array.get(p2_dir, _gi))
                        array.set(_gdIdx,  _gdCachedMaxI, _gi)   // store neighbour index for later
                        // Update cached max
                        _gdCachedMaxD := array.get(_gdDist, 0)
                        _gdCachedMaxI := 0
                        for _jj = 1 to _kGD - 1
                            if array.get(_gdDist, _jj) > _gdCachedMaxD
                                _gdCachedMaxD := array.get(_gdDist, _jj)
                                _gdCachedMaxI := _jj

                // 2. After the neighbour scan, compute mean distance for this regime (sigma update)
                float _meanDist = 0.0
                int _nValid = 0
                for _si = 0 to _kGD - 1
                    if array.get(_gdDist, _si) < 1e8
                        _meanDist += array.get(_gdDist, _si)
                        _nValid   += 1
                if _nValid > 0
                    _meanDist := _meanDist / float(_nValid)
                    float _sigmaNew = math.max(_meanDist / math.sqrt(2.0), 0.1)
                    float _sigmaOld = array.get(sigmaPerRegime, _rfRgm)
